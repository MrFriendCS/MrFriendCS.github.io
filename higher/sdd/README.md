### Abbreviations

asc = Character to ASCII,
chr = ASCII to character,
int = Real to integer,
mod = Modulus,
pa = Parallel arrays,
r = Read from file,
ar = Array of records,
sp = Sub-programs,
sp+ = sp calls a sp,
ss = Sub-strings,
w = Write to file,
ls = Linear search,
ls+ = ls plus a twist,
co = Count occurrences,
co+ = co plus a twist,
min = Find minimum,
min+ = min plus a twist,
max = Find maximum,
max+ = max plus a twist

pc = Pseudocode,
pc+ = pc plus refinements,
sd = Structure diagram
